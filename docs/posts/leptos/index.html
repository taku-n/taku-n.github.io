Panic Hook

nvim src/main.rs

use wasm_bindgen::prelude::*;
use std::panic;

#[wasm_bindgen]
extern "C" {
    #[wasm_bindgen(js_namespace = console)]
    fn log(s: &str);
    #[wasm_bindgen(js_namespace = console)]
    fn error(s: &str);
}

fn panic_hook(panic_info: &panic::PanicInfo) {
    error(&panic_info.to_string());
}

#[wasm_bindgen(start)]
async fn wain() {
    panic::set_hook(Box::new(panic_hook));
    log("hello, world");
    error("Error!");
    panic!("Panic!");
}

fn main() {
}

trunk serve

See a panic message, “Panic!”.
